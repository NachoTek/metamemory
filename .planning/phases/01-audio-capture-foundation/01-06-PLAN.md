```python
def mousePressEvent(self, event):
    """Record press position for click vs drag detection."""
    if event.button() == Qt.MouseButton.LeftButton:
        self.drag_start_pos = event.globalPosition().toPoint()
        self.widget_start_pos = self.pos()
        self.press_time = QTime.currentTime()
        # DON'T accept - let events propagate to child items
        super().mousePressEvent(event)
    else:
        super().mousePressEvent(event)

def mouseReleaseEvent(self, event):
    """Handle click (short duration, small movement) or start drag."""
    if event.button() == Qt.MouseButton.LeftButton:
        release_pos = event.globalPosition().toPoint()
        release_time = QTime.currentTime()

        # Calculate movement distance and time elapsed
        movement = (release_pos - self.drag_start_pos).manhattanLength()
        elapsed_ms = self.press_time.msecsTo(release_time)

        # Threshold: less than 5 pixels and less than 300ms = click, not drag
        if movement < 5 and elapsed_ms < 300:
            # This is a click - let child items handle it
            # Don't start dragging, just pass through
            super().mouseReleaseEvent(event)
        else:
            # This is a drag - start dragging
            if self.is_dragging:
                delta = release_pos - self.drag_start_pos
                new_pos = self.widget_start_pos + delta
                self.move(new_pos)
                self.is_docked = False
                self.dock_edge = None

                # Check edge snapping
                screen = QApplication.primaryScreen().availableGeometry()
                edge_threshold = 20

                if abs(new_pos.x() - screen.left()) < edge_threshold:
                    self.dock_edge = 'left'
                    self.is_docked = True
                    self.move(screen.left(), new_pos.y())
                elif abs(new_pos.x() - screen.right() + self.width()) < edge_threshold:
                    self.dock_edge = 'right'
                    self.is_docked = True
                    self.move(screen.right() - self.width(), new_pos.y())
                elif abs(new_pos.y() - screen.top()) < edge_threshold:
                    self.dock_edge = 'top'
                    self.is_docked = True
                    self.move(new_pos.x(), screen.top())
                elif abs(new_pos.y() - screen.bottom() + self.height()) < edge_threshold:
                    self.dock_edge = 'bottom'
                    self.is_docked = True
                    self.move(new_pos.x(), screen.bottom() - self.height())

                if self.is_docked:
                    self.apply_dock_geometry()
    else:
        super().mouseReleaseEvent(event)
```
